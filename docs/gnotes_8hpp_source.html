70template <RangeOfNoteValues Range> constexpr Signature signature(const Range &intervals) {
71 constexpr Signature mask{1};
72 Unsigned signature{0};
73
74 for (const auto noteValue : intervals | normalize) {
75 auto value = static_cast<Unsigned>(noteValue);
76 signature = signature | (mask << value);
77 }
78
79 return static_cast<Signature>(signature);
80}

331constexpr auto rebaseNoteValue(NoteValue targetBaseNoteValue, NoteValue currentBaseNoteValue) {
332
333 if ((targetBaseNoteValue % NI::Octave != 0) || (currentBaseNoteValue % NI::Octave != 0)) {
334 GTHROW(GInvalidArgument, "Base notes for rebasing must be a C-note: ", noteName(targetBaseNoteValue),
335 noteName(currentBaseNoteValue));
336 }
337
338 auto fn = std::ranges::views::transform([=](NoteValue noteValue) {
339 const auto zeroBasedNoteValue = noteValue - currentBaseNoteValue;
340 return zeroBasedNoteValue + targetBaseNoteValue;
341 });
342 return fn;
343}
