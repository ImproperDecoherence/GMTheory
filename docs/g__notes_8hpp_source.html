85template <RangeOfNoteValues Range> constexpr Signature signature(const Range &intervals) {
86 constexpr Signature mask{1};
87 Unsigned signature{0};
88
89 for (const auto noteValue : intervals | normalize) {
90 auto value = static_cast<Unsigned>(noteValue);
91 signature = signature | (mask << value);
92 }
93
94 return static_cast<Signature>(signature);
95}

346constexpr auto rebaseNoteValue(NoteValue targetBaseNoteValue, NoteValue currentBaseNoteValue) {
347
348 if ((targetBaseNoteValue % NI::Octave != 0) || (currentBaseNoteValue % NI::Octave != 0)) {
349 GTHROW(GInvalidArgument, "Base notes for rebasing must be a C-note: ", noteName(targetBaseNoteValue),
350 noteName(currentBaseNoteValue));
351 }
352
353 auto fn = std::ranges::views::transform([=](NoteValue noteValue) {
354 const auto zeroBasedNoteValue = noteValue - currentBaseNoteValue;
355 return zeroBasedNoteValue + targetBaseNoteValue;
356 });
357 return fn;
358}
